async def init_db() -> None:
    async with aiosqlite.connect(DB_PATH) as db:
        # Таблица заказов (Order)
        await db.execute(
            """
            CREATE TABLE IF NOT EXISTS orders (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id INTEGER NOT NULL,
                product_links TEXT NOT NULL DEFAULT '',
                cost_cny REAL NOT NULL DEFAULT 0,
                weight REAL NOT NULL DEFAULT 0,
                city TEXT NOT NULL DEFAULT '',
                status TEXT NOT NULL,
                comment TEXT NOT NULL DEFAULT '',
                track_code TEXT NOT NULL DEFAULT '',
                cargo_type TEXT NOT NULL DEFAULT '',
                city_from TEXT NOT NULL DEFAULT '',
                phone TEXT NOT NULL DEFAULT '',
                created_at TEXT NOT NULL DEFAULT (datetime('now'))
            )
            """
        )
        # Таблица пользователей (User)
        await db.execute(
            """
            CREATE TABLE IF NOT EXISTS users (
                user_id INTEGER PRIMARY KEY,
                username TEXT,
                full_name TEXT,
                notify_orders INTEGER NOT NULL DEFAULT 1,
                onboarding_done INTEGER NOT NULL DEFAULT 0,
                created_at TEXT NOT NULL DEFAULT (datetime('now'))
            )
            """
        )
        # Таблица уведомлений (Notifications)
        await db.execute(
            """
            CREATE TABLE IF NOT EXISTS notifications (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id INTEGER NOT NULL,
                message TEXT NOT NULL,
                created_at TEXT NOT NULL DEFAULT (datetime('now'))
            )
            """
        )
        # Таблица транзакций/платежей (Payments)
        await db.execute(
            """
            CREATE TABLE IF NOT EXISTS payments (
                payment_id TEXT PRIMARY KEY,
                order_id INTEGER NOT NULL,
                amount REAL NOT NULL,
                currency TEXT NOT NULL DEFAULT 'KZT',
                status TEXT NOT NULL DEFAULT 'pending',
                created_at TEXT NOT NULL DEFAULT (datetime('now'))
            )
            """
        )

Логика расчета стоимости
Файл: utils.py

def commission_percent(cost_cny: float, threshold_cny: float) -> float:
    return 5.0 if cost_cny < threshold_cny else 7.0

def build_calculation(
    cost_cny: float,
    weight_kg: float,
    *,
    cny_rate: float | None = None,
    usd_rate: float | None = None,
    threshold: float,
    tech_fee: float,
    min_kg: float,
    usd_per_kg: float,
    kzt_per_rub: float,
) -> dict:
    import config
    cny = cny_rate if cny_rate is not None else config.CNY_RATE
    usd = usd_rate if usd_rate is not None else config.USD_RATE
    
    # +3% к изначальной цене в юанях
    cost_cny_with_fee = cost_cny * 1.03
    
    goods_rub = round(cost_cny_with_fee * cny, 2)
    pct = commission_percent(cost_cny_with_fee, threshold)
    commission_rub = round(goods_rub * pct / 100, 2)
    del_usd = delivery_usd(weight_kg, min_kg, usd_per_kg)
    delivery_rub = round(del_usd * usd, 2)
    
    import math

    # Итоговая сумма включает всё вместе с фиксированным сбором (tech_fee = 500 руб)
    total_rub = math.ceil(goods_rub + commission_rub + delivery_rub + tech_fee)
    
    return {
        "cny_rate": cny,
        "usd_rate": usd,
        "goods_rub": goods_rub,
        "commission_pct": pct,
        "commission_rub": commission_rub,
        "delivery_usd": del_usd,
        "delivery_rub": delivery_rub,
        "total_rub": total_rub,
        "total_kzt": math.ceil(total_rub * kzt_per_rub),
    }
